# Rust

```
struct Rectangle {
    width: u32,
    height: u32,
    area: u32,
}
impl Rectangle {
    fn calculate_area(&mut self) {
        self.area = self.width * self.height;
    }
}
#[no_mangle]
pub extern fn calc_area(_w: u32, _h: u32) -> u32 {
    let rec = &mut Rectangle {width: _w, height: _h, area: 0};
    rec.calculate_area();
    return rec.area;
}
```
